let currentItem="", currentPrice=0;
let coins=1000;

// BUY
function buy(item,price){
currentItem=item;
currentPrice=price;
document.getElementById("itemName").innerText=item;
document.getElementById("itemPrice").innerText="₹"+price;
document.getElementById("popup").style.display="block";
}

// BUY COINS
function buyCoins(amount){
coins+=amount;
updateBalance();
addHistory("Bought "+amount+" Coins 💎");
showSuccess();
}

// CLOSE
function closePopup(){
document.getElementById("popup").style.display="none";
}

// CONFIRM
function confirmPurchase(){
let user=document.getElementById("username").value;
if(user===""){
alert("Enter username");
return;
}

addHistory(user+" bought "+currentItem+" for ₹"+currentPrice);
closePopup();
showSuccess();
}

// SUCCESS
function showSuccess(){
let box=document.getElementById("successBox");
box.classList.add("show");
setTimeout(()=>box.classList.remove("show"),2000);
}

// HISTORY
function addHistory(text){
let list=document.getElementById("historyList");
let div=document.createElement("div");
div.className="history-item";
div.innerText=text;
list.prepend(div);
}

// BALANCE
function updateBalance(){
document.getElementById("balance").innerText="💎 Coins: "+coins;
}

// PARTICLES
const canvas=document.getElementById("particles");
const ctx=canvas.getContext("2d");
canvas.width=window.innerWidth;
canvas.height=window.innerHeight;

let particles=[];
for(let i=0;i<100;i++){
particles.push({x:Math.random()*canvas.width,y:Math.random()*canvas.height,r:2,speed:0.5});
}

function animate(){
ctx.clearRect(0,0,canvas.width,canvas.height);
particles.forEach(p=>{
p.y+=p.speed;
if(p.y>canvas.height)p.y=0;
ctx.beginPath();
ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
ctx.fillStyle="#22c55e";
ctx.fill();
});
requestAnimationFrame(animate);
}
animate();
